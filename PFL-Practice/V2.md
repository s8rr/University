### 13. Square root of a number

Method 1: Newton-Raphson method (no library function)
```c
#include <stdio.h>
int main() {
    double n, guess;
    scanf("%lf", &n);
    guess = n / 2;
    if (n == 0) guess = 0;
    for (int i = 0; i < 1000 && n != 0; i++)
        guess = (guess + n / guess) / 2;
    printf("Square root = %.4lf\n", guess);
    return 0;
}
```

Method 2: binary search method (no library function)
```c
#include <stdio.h>
int main() {
    double n, low, high, mid;
    scanf("%lf", &n);
    low = 0;
    high = (n < 1) ? 1 : n;
    for (int i = 0; i < 100; i++) {
        mid = (low + high) / 2;
        if (mid * mid > n)
            high = mid;
        else
            low = mid;
    }
    printf("Square root = %.4lf\n", mid);
    return 0;
}
```
